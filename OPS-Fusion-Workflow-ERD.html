// OPS Fusion — Quotation × JO × Sales-Audit (matches prisma/schema/)
Table Customer { id varchar [pk] name varchar tin varchar company varchar }
Table Inquiry {
  id varchar [pk]
  customerId varchar [ref: > Customer.id]
  customerName varchar // free text until firmed at the quotation
  contactNumber varchar
  email varchar // collected by the public portal (reply-to)
  medium varchar // MESSENGER | EMAIL | WALK_IN | CALL | PORTAL
  servicesRequested varchar
  quotationId varchar [unique, ref: - Quotation.id]
}
Table Quotation {
  id varchar [pk]
  quoteNumber varchar [unique] // Q-#### / PO-#### / NJ-####
  type varchar // SALES | PO | NON_JO (discriminator, mirrors JobOrder.isPO/isNonJo)
  poNumber varchar // customer PO reference — type PO only
  customerId varchar [ref: > Customer.id]
  status varchar // DRAFT..SENT..APPROVED..CONVERTED
  taxType varchar // NON_VAT | VAT_EXCLUSIVE | VAT_INCLUSIVE
  taxAmount "decimal(12,2)"
  paymentTermLabel varchar
  downpaymentRate "decimal(3,2)"
  total "decimal(12,2)"
}
Table QuotationItem {
  id varchar [pk]
  quotationId varchar [ref: > Quotation.id]
  productId varchar [ref: > Product.id] // nullable — empty = custom item
  description varchar
  specs json // size, material, color — per product type
  qty int
  unitPrice "decimal(12,2)"
  discount "decimal(12,2)" // per-line discount amount
  lineTotal "decimal(12,2)"
  sortOrder int
}
Table Product { id varchar [pk] name varchar category varchar }
Table PriceRule {
  id varchar [pk]
  productId varchar [ref: > Product.id]
  type varchar // VARIANT | ADDON
  label varchar
  unitPrice "decimal(12,2)" // VARIANT: price per product.unit
  minQty int // VARIANT: qty-tier threshold
  minCharge "decimal(12,2)" // VARIANT: floor for area pricing
  amount "decimal(12,2)" // ADDON: flat fee
  pct "decimal(5,2)" // ADDON: percent of base line
  isActive boolean
}
Table JobOrder {
  id varchar [pk]
  joNumber varchar [unique]
  customerId varchar [ref: > Customer.id]
  quotationId varchar [unique, ref: - Quotation.id] // THE CONVERT LINK
  isPO boolean
  isNonJo boolean
  isApprovedByCustomer boolean
  status varchar
}
Table JobOrderItem {
  id varchar [pk]
  jobOrderId varchar [ref: > JobOrder.id]
  productId varchar [ref: > Product.id] // nullable — legacy free text
  description varchar
  qty int
  unitPrice "decimal(12,2)"
  lineTotal "decimal(12,2)"
  qtyDelivered int // maintained by DR issuance
  productionStatus varchar // "Status - Department"
  deadline timestamp
  waitingPickupSince timestamp
  archivedAt timestamp // done items auto-archive
  lineItemId varchar // "-01"
  isLFP boolean // feeds PRISM later
}
Table DeliveryReceipt {
  id varchar [pk]
  drNumber varchar [unique]
  jobOrderId varchar [ref: > JobOrder.id]
  customerId varchar [ref: > Customer.id]
  bookletId varchar [ref: > Booklet.id]
  isFullDelivery boolean
}
Table DeliveryReceiptLine {
  id varchar [pk]
  deliveryReceiptId varchar [ref: > DeliveryReceipt.id]
  jobOrderItemId varchar [ref: > JobOrderItem.id]
  qty int
}
Table Booklet { id varchar [pk] type varchar seriesStart int seriesEnd int nextNumber int }
Table Sale {
  id varchar [pk]
  documentNo varchar
  bookletId varchar [ref: > Booklet.id]
  customerId varchar [ref: > Customer.id]
  jobOrderId varchar [unique, ref: - JobOrder.id] // null = non-JO sale (NJ quotes land here)
  type varchar // SI_VAT | SI_NON_VAT
  saleDate timestamp
  amount "decimal(12,2)"
  vatAmount "decimal(12,2)"
}
Table CollectionReceipt {
  id varchar [pk]
  crNumber varchar [unique]
  bookletId varchar [ref: > Booklet.id]
  customerId varchar [ref: > Customer.id]
  amount "decimal(12,2)"
  method varchar // CASH | CHECK | GCASH | BANK_TRANSFER | QR
  receivedAt timestamp
}
Table CrAllocation {
  id varchar [pk]
  crId varchar [ref: > CollectionReceipt.id]
  saleId varchar [ref: > Sale.id]
  amount "decimal(12,2)"
}
Table AdvancePayment {
  id varchar [pk]
  customerId varchar [ref: > Customer.id]
  amount "decimal(12,2)"
  method varchar // CASH | CHECK | GCASH | BANK_TRANSFER | QR
  status varchar // UNAPPLIED | PARTIALLY_APPLIED | FULLY_APPLIED
  reference varchar // OR / deposit slip / GCash ref
  receivedAt timestamp
}
Table AdvancePaymentApplication {
  id varchar [pk]
  advancePaymentId varchar [ref: > AdvancePayment.id]
  jobOrderId varchar [ref: > JobOrder.id]
  deliveryReceiptId varchar [ref: > DeliveryReceipt.id]
  amount "decimal(12,2)"
}
Table AuditEntry {
  id varchar [pk]
  saleId varchar [ref: > Sale.id]
  status varchar
  flagType varchar
}
Table ReconciliationDay { id varchar [pk] date date [unique] status varchar }

The OPS Fusion ruling — decided

Per your "ikaw bahala" — here's the flow I'm locking in, with the one seam that needs the collaborator.

W2 Quotation-first is the DEFAULT path, not a hard lock.

Formal jobs go Inquiry → Quotation → Convert. But the legacy shop also took walk-ins straight to JO and counter sales with no JO at all — both stay: direct JO entry (already built) and non-JO Sale (jobOrderId = null). Removing them would break "legacy 1:1".

W3 One Customer master, firmed at the quotation.

The client on the quote IS the customer on the JO, the Sale, the CR, and the DR — one row, one FK everywhere. Portal inquiries carry free-text names; the record is picked/created when the quotation is made (already how the combobox works).

W4 Convert is the only bridge — and it's transactional.

One click on an APPROVED/SENT quote creates the DRAFT JO (items copied 1:1, totals carried, history stamped, quote → CONVERTED) — all-or-nothing in one transaction. No copy-paste hand-off, no double conversion (unique quotationId guards it at the database level). The JO's own approval gate still applies before production.

W1 ✓ DONE — the type IS picked up-front, at the quotation.

Quotation.type (SALES / PO / NON_JO) + poNumber are live, with type-aware quote numbering (Q- / PO- / NJ-). Exactly the "mag-select na daan" rule — implemented one step earlier than the Convert dialog originally proposed, which is better.

W1b REMAINING SEAM — Convert must honor the quote's type (collaborator task).

Today convertToJobOrder still creates a plain auto-numbered JO regardless of type. Two fixes: (1) type PO → set isPO = true and carry poNumber as the JO's manual number; (2) type NON_JO → block conversion (an NJ quote is billed as a Sale with jobOrderId = null, never produced). Small service change, no schema change.

W5 SEAM ALREADY MARKED — advance payments activate at the Sales merge.

Downpayment (per the quote's downpaymentRate) is recorded as an AdvancePayment on the Sales side, can gate the JO approval, and applies at DR/Sale. The models and TODO markers exist; wiring happens when Sales-Audit lands.
